## The engine ↔ platform bridge (single source of truth)

`platform.js` (module, deferred — runs after the inlined bundle) calls:

```js
const stage = CASINO.boot.start({ model, ui });
```

- `model` = `buildFloorModel()` result (§Task 2).
- `ui` = callbacks the ENGINE calls (implemented in platform.js/ui.js):
  - `ui.onNear(anchor|null)` — anchor = `{ kind:'table'|'cashier'|'reception'|'practice', id, table? }` (table = floor-model table entry). Fired when nearest-in-radius changes.
  - `ui.onSectionChange(zoneId|null)` — `'vestibule'|'roulette'|'blackjack'|'baccarat'|'uth'|'aisle'`.
  - `ui.onTurnstileBlocked()` — player pushed against the closed turnstile (throttled ≥1.5 s).
  - `ui.onDeskReached()` — camera arrived at the counter POV after `walkToDesk()`.
- `stage` = engine/floor API platform calls:
  - `stage.setAccess(open:boolean)` — turnstile open/green vs closed/red (tweened; instant if REDUCED).
  - `stage.walkToDesk() → Promise` — glide to counter POV.
  - `stage.playWelcome({ name }) → Promise` — stamp animation on the member card (instant if REDUCED).
  - `stage.playHeadShake() → Promise` — receptionist refusal.
  - `stage.playWaveThrough() → Promise` — ~1.5 s pass-by greeting, click-to-skip, ends ON the floor.
  - `stage.goTo(anchorId) → Promise` — `'spawn'|'reception'|'floor'|'cashier'|'practice'|'roulette'|'blackjack'|'baccarat'|'uth'` glide (instant if REDUCED).
  - `stage.setHighlight(tableId|null)` — table glow rig on/off. `tableId = section + ':' + (key ?? 'main')`.

Anchors/obstacles are registered by floor builders into `C.world` (created by `layout.js`):
`C.world.addAnchor({ id, kind, pos:[x,z], radius, approach:{pos:[x,y,z], look:[x,y,z]}, table? })`,
`C.world.addObstacle({ x, z, r })`, `C.world.zones` (rect list with ids).

```js
// floor-model.js — pure registry: GAME_STAKES → the 3D floor's sections and
// physical stake tables. Imported at runtime by platform.js and by node tests.
// Parity with the 2D stake pickers is by construction (same table-config).
import { gameStakes, getTable } from '../js/wallet/table-config.js';

export const SECTION_ORDER = ['roulette', 'blackjack', 'baccarat', 'uth'];

const GAME_META = {
  roulette:  { label: 'ROULETTE',         gameName: 'Roulette',                base: '../roulette/index.html' },
  blackjack: { label: 'BLACKJACK',        gameName: 'Blackjack',               base: '../blackjack/game-mode/index.html' },
  baccarat:  { label: 'BACCARAT',         gameName: 'Baccarat',                base: '../baccarat/game-mode/index.html' },
  uth:       { label: "ULTIMATE HOLD'EM", gameName: "Ultimate Texas Hold'em",  base: '../ultimate-texas-holdem/index.html' },
};

// Smallest stake the table accepts — drives the sit-down card's insufficient hint.
export function tableMinBet(table) {
  return Math.min(...Object.values(table.betTypes).map((b) => b.min));
}

export function buildFloorModel() {
  const sections = SECTION_ORDER.map((game) => {
    const meta = GAME_META[game];
    if (game === 'uth') {
      return {
        id: game, label: meta.label, gameName: meta.gameName, reservedSpots: 4,
        tables: [{
          id: 'uth:main', key: null, gameId: 'uth', tierName: 'Table 1',
          limitsText: 'Ante 100 – 1,000 · Buy-in 10,000',
          blurb: 'Online multiplayer vs the dealer.',
          href: meta.base, minBet: 10000,
        }],
      };
    }
    return {
      id: game, label: meta.label, gameName: meta.gameName, reservedSpots: 0,
      tables: (gameStakes(game) || []).map((s) => ({
        id: `${game}:${s.key}`, key: s.key, gameId: s.gameId, tierName: s.name,
        limitsText: s.limitsText, blurb: s.blurb,
        href: `${meta.base}?stake=${s.key}`,
        minBet: tableMinBet(getTable(s.gameId)),
      })),
    };
  });
  return { sections };
}
```

```js
// reception-model.js — pure state machine for the vestibule ID check.
// platform.js dispatches events from onAuth/DOM/animation completions and
// renders phases into the turnstile, camera moves, and the check-in card.
export const initialReception = { phase: 'boot', error: null };

export function receptionReduce(state, ev) {
  switch (ev.type) {
    case 'AUTH_READY':
      return state.phase === 'boot'
        ? { phase: ev.signedIn ? 'wave' : 'out', error: null } : state;
    case 'AUTH_UNAVAILABLE': return { phase: 'unavailable', error: 'auth-unavailable' };
    case 'OPEN_CHECKIN':  return state.phase === 'out' ? { phase: 'checkin', error: state.error } : state;
    case 'CLOSE_CHECKIN': return state.phase === 'checkin' ? { phase: 'out', error: null } : state;
    case 'SIGNIN_START':  return state.phase === 'checkin' ? { phase: 'authing', error: null } : state;
    case 'SIGNIN_ERROR':  return state.phase === 'authing' ? { phase: 'checkin', error: ev.code || 'signin-failed' } : state;
    case 'SIGNED_IN':
      // Auth can also complete from another tab while the card is closed.
      return ['authing', 'checkin', 'out'].includes(state.phase) ? { phase: 'welcome', error: null } : state;
    case 'WELCOME_DONE': return state.phase === 'welcome' ? { phase: 'floor', error: null } : state;
    case 'WAVE_DONE':    return state.phase === 'wave' ? { phase: 'floor', error: null } : state;
    case 'SIGNED_OUT':   return state.phase === 'boot' ? state : { phase: 'out', error: null };
    default: return state;
  }
}
```

```js
(() => {
  const C = (globalThis.CASINO ??= {});
  const frameHooks = new Set();
  const pickables = [];
  const REDUCED = matchMedia('(prefers-reduced-motion: reduce)').matches;
  const IS_MOBILE = matchMedia('(max-width: 768px)').matches;
  const P = { x: -25.5, z: 0, yaw: Math.PI / 2, pitch: 0 }; // yaw π/2 faces +X (east)
  const SPEED = 3.2, GLIDE = 5.0, RADIUS = 0.35;
  const keys = Object.create(null);
  let ui = null, glide = null, lastBlocked = 0;

  const to2D = () => location.replace('../index.html?toast=3d-unavailable');

  function clampWalk(x, z) {
    // inside SOME walk rect (gate rect only when open), outside ALL obstacles
    const rects = C.floorplan.WALK_RECTS.filter((r) => r.id !== 'gate' || C.world.gateOpen);
    const inRect = (r, px, pz) => px > r.x0 + RADIUS && px < r.x1 - RADIUS && pz > r.z0 + RADIUS && pz < r.z1 - RADIUS;
    // the gate/vestibule/floor rects overlap at their seams; accept if any contains
    let ok = rects.some((r) => inRect(r, x, z));
    if (!ok) return null;
    for (const o of C.world.obstacles) {
      const dx = x - o.x, dz = z - o.z, d2 = dx * dx + dz * dz, min = o.r + RADIUS;
      if (d2 < min * min) {
        const d = Math.sqrt(d2) || 1e-4;
        x = o.x + (dx / d) * min; z = o.z + (dz / d) * min;
        if (!rects.some((r) => inRect(r, x, z))) return null;
      }
    }
    return { x, z };
  }

  function tryMove(nx, nz) {
    let r = clampWalk(nx, nz);
    if (!r) r = clampWalk(nx, P.z);      // axis-separated slide
    if (!r) r = clampWalk(P.x, nz);
    if (r) { P.x = r.x; P.z = r.z; return true; }
    // pushed against the closed turnstile?
    if (!C.world.gateOpen && Math.abs(P.x - -16.2) < 1.3 && Math.abs(P.z) < 1.6 && nx > P.x) {
      const now = performance.now();
      if (now - lastBlocked > 1500) { lastBlocked = now; ui?.onTurnstileBlocked?.(); }
    }
    return false;
  }

  C.app = {
    scene: null, camera: null, renderer: null, player: P,
    REDUCED, IS_MOBILE,

    init(opts = {}) {
      ui = opts.ui || null;
      if (new URLSearchParams(location.search).has('nowebgl')) return to2D();
      const canvas = document.getElementById('stage');
      let renderer;
      try {
        THREE.ColorManagement.legacyMode = false; // r149 sRGB gotcha (v1 comment)
        renderer = new THREE.WebGLRenderer({ canvas, antialias: !IS_MOBILE });
        renderer.outputEncoding = THREE.sRGBEncoding;
      } catch (err) { console.error('WebGL init failed:', err); return to2D(); }
      canvas.addEventListener('webglcontextlost', (e) => { e.preventDefault(); to2D(); });
      renderer.setPixelRatio(Math.min(devicePixelRatio, IS_MOBILE ? 1.5 : 2));
      renderer.setSize(innerWidth, innerHeight);
      renderer.shadowMap.enabled = !IS_MOBILE;
      renderer.shadowMap.type = THREE.PCFSoftShadowMap;
      const scene = new THREE.Scene();
      const camera = new THREE.PerspectiveCamera(58, innerWidth / innerHeight, 0.05, 120);
      Object.assign(C.app, { renderer, scene, camera });
      addEventListener('resize', () => {
        camera.aspect = innerWidth / innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(innerWidth, innerHeight);
      });

      // ----- input: drag-look vs tap -----
      const ray = new THREE.Raycaster(), ptr = new THREE.Vector2();
      const floorPlane = new THREE.Plane(new THREE.Vector3(0, 1, 0), 0);
      let down = null, dragging = false;
      canvas.addEventListener('pointerdown', (e) => { down = { x: e.clientX, y: e.clientY, yaw: P.yaw, pitch: P.pitch }; dragging = false; });
      canvas.addEventListener('pointermove', (e) => {
        if (!down) return;
        const dx = e.clientX - down.x, dy = e.clientY - down.y;
        if (!dragging && Math.hypot(dx, dy) > 7) dragging = true;
        if (dragging && e.pointerType === 'mouse') {
          P.yaw = down.yaw - dx * 0.004;
          P.pitch = Math.max(-0.5, Math.min(0.4, down.pitch - dy * 0.003));
        }
      });
      canvas.addEventListener('pointerup', (e) => {
        const wasDrag = dragging; down = null; dragging = false;
        if (wasDrag) return;
        ptr.set((e.clientX / innerWidth) * 2 - 1, -(e.clientY / innerHeight) * 2 + 1);
        ray.setFromCamera(ptr, camera);
        const hit = ray.intersectObjects(pickables.map((p) => p.mesh), true)[0];
        if (hit) {
          let obj = hit.object;
          while (obj) { const f = pickables.find((p) => p.mesh === obj); if (f) return f.onClick(); obj = obj.parent; }
        }
        const pt = new THREE.Vector3();
        if (ray.ray.intersectPlane(floorPlane, pt)) C.app.walkTo(pt.x, pt.z);
      });
      addEventListener('keydown', (e) => {
        if (e.target instanceof HTMLElement && ['INPUT', 'BUTTON', 'A', 'TEXTAREA'].includes(e.target.tagName)) return;
        keys[e.code] = true;
      });
      addEventListener('keyup', (e) => { keys[e.code] = false; });

      // ----- frame loop -----
      let last = performance.now(), proxTimer = 0, nearId; let zoneId;
      const loop = (now) => {
        const dt = Math.min(0.05, (now - last) / 1000); last = now;
        // WASD (cancels glide)
        const f = (keys.KeyW ? 1 : 0) - (keys.KeyS ? 1 : 0);
        const s = (keys.KeyD ? 1 : 0) - (keys.KeyA ? 1 : 0);
        if (f || s) {
          glide = null;
          const len = Math.hypot(f, s) || 1;
          const dx = (Math.sin(P.yaw) * f + Math.cos(P.yaw) * s) / len;
          const dz = (Math.cos(P.yaw) * f - Math.sin(P.yaw) * s) / len;
          tryMove(P.x + dx * SPEED * dt, P.z + dz * SPEED * dt);
        } else if (glide) {
          const dx = glide.x - P.x, dz = glide.z - P.z, d = Math.hypot(dx, dz);
          if (d < 0.05) { const done = glide.onDone; glide = null; done?.(); }
          else {
            tryMove(P.x + (dx / d) * Math.min(GLIDE * dt, d), P.z + (dz / d) * Math.min(GLIDE * dt, d));
            if (glide && glide.faceTarget) {
              const want = Math.atan2(dx, dz);
              let dy = want - P.yaw; while (dy > Math.PI) dy -= 2 * Math.PI; while (dy < -Math.PI) dy += 2 * Math.PI;
              P.yaw += dy * Math.min(1, dt * 6);
            }
          }
        }
        // proximity + zone at 10 Hz
        proxTimer += dt;
        if (proxTimer > 0.1) {
          proxTimer = 0;
          let best = null, bestD = Infinity;
          for (const a of C.world.anchors) {
            const d = Math.hypot(P.x - a.pos[0], P.z - a.pos[1]);
            if (d < a.radius && d < bestD) { best = a; bestD = d; }
          }
          if ((best && best.id) !== nearId) { nearId = best && best.id; ui?.onNear?.(best || null); }
          const zone = C.floorplan.ZONES.find((zz) => P.x >= zz.x0 && P.x <= zz.x1 && P.z >= zz.z0 && P.z <= zz.z1);
          if ((zone && zone.id) !== zoneId) { zoneId = zone && zone.id; ui?.onSectionChange?.(zoneId || null); }
        }
        camera.position.set(P.x, C.floorplan.EYE, P.z);
        camera.lookAt(P.x + Math.sin(P.yaw) * Math.cos(P.pitch), C.floorplan.EYE + Math.sin(P.pitch), P.z + Math.cos(P.yaw) * Math.cos(P.pitch));
        frameHooks.forEach((fn) => fn(dt, now / 1000));
        renderer.render(scene, camera);
        requestAnimationFrame(loop);
      };
      requestAnimationFrame(loop);
    },

    walkTo(x, z, onDone) {
      const c = clampWalk(x, z) || { x, z };
      if (REDUCED) { tryMove(c.x, c.z) || Object.assign(P, { x: c.x, z: c.z }); onDone?.(); return; }
      glide = { x: c.x, z: c.z, faceTarget: true, onDone };
    },

    // fly the camera to an exact pose (position+look), Promise-based, for stage.goTo
    glideTo(pos, look, ms = 1100) {
      return new Promise((res) => {
        const wantYaw = Math.atan2(look[0] - pos[0], look[2] - pos[2]);
        const wantPitch = Math.atan2(look[1] - C.floorplan.EYE, Math.hypot(look[0] - pos[0], look[2] - pos[2])) * 0.9;
        if (REDUCED) { Object.assign(P, { x: pos[0], z: pos[2], yaw: wantYaw, pitch: wantPitch }); return res(); }
        const from = { ...P };
        let dyaw = wantYaw - from.yaw; while (dyaw > Math.PI) dyaw -= 2 * Math.PI; while (dyaw < -Math.PI) dyaw += 2 * Math.PI;
        C.tween.to(P, {}, ms); // noop keeps API symmetric; drive manually below
        const t0 = performance.now();
        const hook = () => {
          const t = Math.min(1, (performance.now() - t0) / ms);
          const e = C.tween.easings.inOutCubic(t);
          P.x = from.x + (pos[0] - from.x) * e; P.z = from.z + (pos[2] - from.z) * e;
          P.yaw = from.yaw + dyaw * e; P.pitch = from.pitch + (wantPitch - from.pitch) * e;
          if (t >= 1) { C.app.offFrame(hook); res(); }
        };
        hook.cancel = () => { C.app.offFrame(hook); res(); };
        glide = null; C.app.onFrame(hook);
      });
    },
    goToAnchor(a) { return C.app.glideTo(a.approach.pos, a.approach.look); },
    spawn() {
      const s = C.floorplan.SPAWN;
      P.x = s.pos[0]; P.z = s.pos[2];
      P.yaw = Math.atan2(s.look[0] - s.pos[0], s.look[2] - s.pos[2]); P.pitch = 0;
    },

    onFrame: (fn) => frameHooks.add(fn),
    offFrame: (fn) => frameHooks.delete(fn),
    addPickable: (mesh, onClick) => pickables.push({ mesh, onClick }),

    banner(title, sub = '', ms = 2600) { /* keep v1 implementation verbatim */ },
    fade(midFn) { /* keep v1 implementation verbatim */ },
  };
})();
```
